``` python
class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:
        if not matrix: return []
        left, right, top, down = 0, len(matrix[0]) - 1, 0, len(matrix) - 1
        res = []
        while left <= right and top <= down:
            # left to right
            res += matrix[top][left:right+1]
            top += 1
            # top to down
            res += [matrix[i][right] for i in range(top, down+1)]
            right -= 1
            # right to left 
            if top <= down:
                res += matrix[down][left:right+1][::-1]
                down -= 1
            # bottom to up
            if left <= right:
                res += [matrix[i][left] for i in range(top, down+1)][::-1]
                left += 1
        return res 
```
